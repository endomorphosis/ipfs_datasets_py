1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "tdfol.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "tdfol.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "tdfol.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "tdfol.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

2. Monitor the deployment:
  kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "tdfol.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -w

3. View logs:
  kubectl logs --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "tdfol.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -f

4. Check service health:
  {{- if .Values.ingress.enabled }}
  curl -f {{ (index .Values.ingress.hosts 0).host }}/health
  {{- else }}
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "tdfol.fullname" . }} 8080:{{ .Values.service.port }}
  curl -f http://localhost:8080/health
  {{- end }}

5. Access Prometheus metrics:
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "tdfol.fullname" . }} 9090:{{ .Values.service.metricsPort }}
  curl http://localhost:9090/metrics

{{- if .Values.redis.enabled }}

6. Redis connection:
  Redis is available at: {{ include "tdfol.redis.fullname" . }}:6379
  To connect from a pod:
    redis-cli -h {{ include "tdfol.redis.fullname" . }} -p 6379
{{- end }}

{{- if .Values.autoscaling.enabled }}

7. Horizontal Pod Autoscaler:
  kubectl get hpa --namespace {{ .Release.Namespace }} {{ include "tdfol.fullname" . }} -w
{{- end }}
